In ECMAScript 6 (ES6) modules, the __dirname and __filename variables are not available as they are 
in CommonJS modules. Instead, you can use the import.meta object to access similar information.

import { fileURLToPath } from 'url';
import { dirname } from 'path';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

console.log(__dirname);
